//基于链表的栈，从头结点入栈，出栈，头结点中的val存储长度
typedef struct Node{
    int val;
    struct Node *next;
}Node;
Node *getStack(){
    Node *head = (Node *)malloc(sizeof(Node));
    head->next = NULL;
    head->val = 0;
    return head;
}
int push(Node *head, int val){
    if(head == NULL){
        return 0;
    }
    Node *p = (Node *)malloc(sizeof(Node));
    p->val = val;
    p->next = head->next;
    head->next = p;
    head->val++;
    return 1;
}
int peek(Node *head){
    if(head == NULL){
        return -1;
    }
    return head->next->val;
}
int pop(Node *stack){
    if(stack == NULL){
        return 0;
    }
    Node *p = stack->next;
    stack->next = stack->next->next;
    stack->val--;
    free(p);
    return 1;
}
void clear(Node *stack){
    if(stack == NULL){
        return ;
    }
    for(Node *p = stack, q; p; p = q){
        q = p->next;
        free(p);
    }
}
int getSize(Node *head){
    if(head == NULL){
        return -1;
    }
    return head->val;
