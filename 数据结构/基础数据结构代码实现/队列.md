/*
一种先进先出的数据结构，且只能在一端进，在另一端出。基于这种特性，结构体中需要首尾指针来确定位置
定义结构：
getQueue()：获取队列，并进行初始化
push()：向队列中加入数据
pop()：将队首的数据返回并清除
getSize()：返回队列中数据的个数
printfQueue()：对队列中的数据进行打印
clear()：销毁栈
*/
//基于数组的队列
#include<stdio.h>
#include<stdlib.h>
//这个N可以放入结构体中，这样更合理
#define N 10
typedef struct Queue{
    int *data;
    int size;
    int front, rear;
}Queue;
Queue *getQueue(){
    Queue *queue = (Queue *)malloc(sizeof(Queue));
    queue->data = (int *)malloc(sizeof(int) * N);
    queue->front = queue->rear = queue->size = 0;
    return queue;
}
int push(Queue *queue, int val){
    //因为是数组构成，所以需要判断N，可以写为动态数组（给个判断，然后重新分配大小）
    //这里也可以看出有个问题，就是会出现假满的情况（在栈“满”的时候其实还有空间）
    if(queue == NULL || queue->rear == N){
        return 0;
    }
    queue->data[(queue->rear)++] = val;
    queue->size++;
    return 1;
}
int pop(Queue *queue){
    if(queue == NULL || queue->size == 0){
        return -1;
    }
    int res = queue->data[queue->front];
    queue->front++;
    queue->size--;
    return res;
}
int getSize(Queue *queue){
    if(queue == NULL){
        return -1;
    }
    return queue->size;
}
void printfQueue(Queue *queue){
    if(queue == NULL){
		return ;
    }
    for(int i = queue->front; i < queue->rear; i++){
		printf("%d ", queue->data[i]);
    }
}
void clear(Queue *queue){
    if(queue == NULL){
        return ;
    }
    free(queue->data);
    free(queue);
}
int main(void){
    Queue *queue = getQueue();
    for(int i = 0; i < 10; i++){
        push(queue, i);
    }
    printf("栈的长度为：%d\n", getSize(queue));
    printfQueue(queue);
    printf("\n");
    for(int i = 0; i < 3; i++){
        printf("%d ", pop(queue));
    }
    printf("\n");
    printfQueue(queue);
    printf("\n");
    printf("栈的长度为：%d", getSize(queue));
	clear(queue);
    return 0;
}

//定义Node表示队列中的结点，将其放入Queue结构体中并声明首尾指针
typedef struct Node{
	int val;
	struct Node *next;
}Node;
typedef struct Queue{
	Node *front, *rear;
	int size;
}Queue;
Queue *getQueue(){
	Queue *queue = (Queue *)malloc(sizeof(Queue));
	queue->size = 0;
	queue->rear = queue->front = NULL;
	return queue;
}
int push(Queue *queue, int val){
	if(queue == NULL){
		return 0;
	}
	Node *node = (Node *)malloc(sizeof(Node));
	node->val = val;
	node->next = NULL;
	if(queue->front == NULL){
		queue->front = node;
		queue->rear = node;
	}else{
		node->next = queue->rear->next;
		queue->rear->next = node;
		queue->rear = node;
	}
	queue->size++;
	return 1;
}
int pop(Queue *queue){
	if(queue == NULL || queue->size == 0){
		return 0;
	}
	if(queue->size > 0){
		queue->front = queue->front->next;
		queue->size--;
		return 1; 
	}
	return 0;
}
int getSize(Queue *queue){
	if(queue == NULL){
		return -1;
	}
	return queue->size;
}

	if(queue == NULL){
		return ;
	}
	Node *p = queue->front;

//基于数组的循环双向列表
typedef struct Queue{
	int *data;
	int front, rear;
	int size;
	int queSize;
}Queue;
Queue *getQueue(int val){
	Queue *queue = (Queue *)malloc(sizeof(Queue));
	queue->size = val;
	queue->data = (int *)malloc(sizeof(int) * queue->size);
	queue->front = queue->rear = queue->queSize = 0;
	return queue;
}
int pushFront(Queue *queue, int val){
	if(queue->size == queue->queSize){
		return 0;
	}
	//在这里因为可能是左右端都超出size范围，所以需要加上size再取余 
	queue->data[((--queue->front) + queue->size) % queue->size] = val;
	queue->queSize++; 
	return 1;
}
int pushRear(Queue *queue, int val){
	if(queue->size == queue->queSize){
		return 0;
	}
	queue->data[((queue->rear++) + queue->size) % queue->size] = val;
	queue->queSize++;
	return 1;
}
int popFront(Queue *queue){
	if(queue == NULL || queue->queSize == 0){
		return 0;
	}
	queue->queSize--;
	return queue->data[((queue->front++) + queue->size) % queue->size];
}
int popRear(Queue *queue){
	if(queue == NULL || queue->queSize == 0){
		return 0;
	}
	queue->queSize--;
	return queue->data[((queue->rear--) + queue->size) % queue->size];
}
int getSize(Queue *queue){
	if(queue == NULL){
		return -1;
	}
	return queue->queSize;
}
void printfQueue(Queue *queue){
	if(queue == NULL){
		return ;
	}
	for(int i = 0; i < queue->queSize; i++){
		printf("%d ", queue->data[i]);
	}
}
int getFront(Queue *queue){
	if(queue == NULL){
		return -1;
	}
	return (queue->front + queue->size) % queue->size;
}
int getRear(Queue *queue){
	if(queue == NULL){
		return -1;
	}
	return (queue->rear + queue->size) % queue->size;
}
void clear(Queue *queue){
	if(queue == NULL){
		return ;
	}
	free(queue->data);
	free(queue);
}
int main(void){

		pushFront(queue, i);
	}
	for(int i = 0; i < 5; i++){

//基于链表的双向队列
typedef struct Node{
    int val;
    struct Node *front;
    struct Node *rear;
}Node;
typedef struct Queue{
    Node *head;
    Node *tail;
    int size;
}Queue;
Queue *getQueue(){
    Queue *queue = (Queue *)malloc(sizeof(Queue));
    queue->size = 0;
    queue->head = NULL;
    queue->tail = NULL;
    return queue;
}
int pushFront(Queue *queue, int val){
    if(queue == NULL){
        return 0;
    }
    Node *node = (Node *)malloc(sizeof(Node));
    node->val = val;
    if(queue->head == NULL){
        node->rear = NULL;
        node->front = NULL;
        queue->tail = node;
        queue->head = node;
    }else{
    	node->front = NULL;
    	node->rear = queue->head;
    	queue->head->front = node;
    	queue->head = node;
	}
	queue->size++;
	return 1;
}
int pushRear(Queue *queue, int val){
	if(queue == NULL){
		return 0;
	}
	Node *node = (Node *)malloc(sizeof(Node));
	node->val = val;
	if(queue->tail == NULL){
		node->rear = NULL;
		node->front = NULL;
		queue->head = node;
		queue->tail = node;
	}else{
		node->rear = NULL;
		queue->tail->rear = node;
		node->front = queue->tail;
		queue->tail = node;
	}
	queue->size++;
	return 1;
}
int popFront(Queue *queue){
	 if (queue == NULL || queue->head == NULL) {
        return -1; // 队列为空或无效，返回错误
    }
    Node *temp = queue->head; // 保存当前头节点
    int value = temp->val;     // 获取头节点的值

    queue->head = queue->head->rear; // 更新头指针

    if (queue->head != NULL) { // 如果新的头不为空，更新前驱指针
        queue->head->front = NULL;
    } else { 
        queue->tail = NULL; // 如果队列变为空，更新尾指针
    }

    free(temp); // 释放原头节点内存
    queue->size--; // 减少队列大小
    return value; //
}
int popRear(Queue *queue){
	 if (queue == NULL || queue->tail == NULL) {
        return -1;
    }
    Node *temp = queue->tail; 
    int value = temp->val;   
    queue->tail = queue->tail->front; 
    if (queue->tail != NULL) { 
        queue->tail->rear = NULL;
    } else { 
        queue->head = NULL; 
    }

    free(temp); 
    queue->size--;
    return value; 
}
void printfQueue(Queue *queue){
	if(queue == NULL || queue->head == NULL){
		return ;
	}
	Node *p = queue->head;
	while(p){
		printf("%d ", p->val);
		p = p->rear;
	}
	printf("\n");
}
void clear(Queue *queue){
    if(queue == NULL){
        return ;
    }
    Node *p = queue->head->rear;
    Node *q;
    while(p){
        q = p->rear;
        free(p);
        p = q;
    }
    free(queue);
}
int main(void){
	Queue *queue = getQueue();
	for(int i = 0; i < 10; i++){
		pushRear(queue, i);
	}
	printf("\n");
	printfQueue(queue);
	printf("%d ", popFront(queue));
	printf("%d ", popFront(queue));
	printf("\n");
	for(int i = 0; i < 2; i++){
		printf("%d ", popRear(queue));
	}
	printf("\n");
	printfQueue(queue);
	clear(queue);
	return 0;
}
