/*
一种先进先出的数据结构，且只能在一端进，在另一端出。基于这种特性，结构体中需要首尾指针来确定位置
定义结构：
getQueue()：获取队列，并进行初始化
push()：向队列中加入数据
pop()：将队首的数据返回并清除
getSize()：返回队列中数据的个数
printfQueue()：对队列中的数据进行打印
clear()：销毁栈
*/
//基于数组的队列
#include<stdio.h>
#include<stdlib.h>
#define N 10
typedef struct Queue{
    int *data;
    int size;
    int front, rear;
}Queue;
Queue *getQueue(){
    Queue *queue = (Queue *)malloc(sizeof(Queue));
    queue->data = (int *)malloc(sizeof(int) * N);
    queue->front = queue->rear = queue->size = 0;
    return queue;
}
int push(Queue *queue, int val){
    //因为是数组构成，所以需要判断N，可以写为动态数组（给个判断，然后重新分配大小）
    //这里也可以看出有个问题，就是会出现假满的情况（在栈“满”的时候其实还有空间）
    if(queue == NULL || queue->rear == N){
        return 0;
    }
    queue->data[(queue->rear)++] = val;
    queue->size++;
    return 1;
}
int pop(Queue *queue){
    if(queue == NULL){
        return -1;
    }
    int res = queue->data[queue->front];
    queue->front++;
    queue->size--;
    return res;
}
int getSize(Queue *queue){
    if(queue == NULL){
        return -1;
    }
    return queue->size;
}
void printfQueue(Queue *queue){
    if(queue == NULL){
		return ;
    }
    for(int i = queue->front; i < queue->rear; i++){
		printf("%d ", queue->data[i]);
    }
}
void clear(Queue *queue){
    if(queue == NULL){
        return ;
    }
    free(queue->data);
    free(queue);
}
int main(void){
    Queue *queue = getQueue();
    for(int i = 0; i < 10; i++){
        push(queue, i);
    }
    printf("栈的长度为：%d\n", getSize(queue));
    printfQueue(queue);
    printf("\n");
    for(int i = 0; i < 3; i++){
        printf("%d ", pop(queue));
    }
    printf("\n");
    printfQueue(queue);
    printf("\n");
    printf("栈的长度为：%d", getSize(queue));
	clear(queue);
    return 0;
}

//怎样循环呢，可以通过取余，只要rear下一个不是front则可，实现方式有多种，按自己喜欢的来理解就行

	int *data;

	int front, rear;
}Queue;
Queue *getQueue(){
	Queue *queue = (Queue *)malloc(sizeof(Queue));
	queue->data = (int *)malloc(sizeof(int) * N);
	queue->front = 0;
	queue->rear = 0;
	queue->size = 0;
	return queue;
} 
int push(Queue *queue, int val){
	if(queue->size == N){
		return 0;
	}
	queue->data[(queue->rear++) % N] = val;
	queue->size++;
	return 1;
}
int pop(Queue *queue){
	if(queue == NULL){
		return -1;
	}
	queue->size--;
	return queue->data[(queue->front++) % N];
	
}

	if(queue == NULL){
		return -1;
	}
	return queue->size;
}
void printfQueue(Queue *queue){
	if(queue == NULL){
		return ;
	}
	for(int i = 0; i < queue->size; i++){

	}
}

	if(queue == NULL){
		return ;
	}
	free(queue->data);
	free(queue);
}

//上面的都是基于数组实现，同样可以是通过链表进行实现
//从头结点后进入队列，在链表末进行出队列
//定义Node表示队列中的结点，将其放入Queue结构体中并声明首尾指针
typedef struct Node{
	int val;
	struct Node *next;
}Node;
typedef struct Queue{
	Node *front, *rear;
	int size;
}Queue;
Queue *getQueue(){
	Queue *queue = (Queue *)malloc(sizeof(Queue));
	queue->size = 0;
	queue->rear = queue->front = NULL;
	return queue;
}
int push(Queue *queue, int val){
	if(queue == NULL){
		return 0;
	}
	Node *node = (Node *)malloc(sizeof(Node));
	node->val = val;
	node->next = NULL;
	if(queue->front == NULL){
		queue->front = node;
		queue->rear = node;
	}else{
		node->next = queue->rear->next;
		queue->rear->next = node;
		queue->rear = node;
	}
	queue->size++;
	return 1;
}
int pop(Queue *queue){
	if(queue == NULL){
		return 0;
	}
	if(queue->size > 0){
		queue->front = queue->front->next;
		queue->size--;
		return 1; 
	}
	return 0;
}
int getSize(Queue *queue){
	if(queue == NULL){
		return -1;
	}
	return queue->size;
}
void printfQueue(Queue *queue){
	if(queue == NULL){
		return ;
	}
	Node *p = queue->front;
	while(p){
		printf("%d ", p->val);
		p = p->next;
	}
}
void clear(Queue *queue){
	if(queue == NULL){
		return ;
	}
	Node *p = queue->front;,
    Node *q;
	while(p){
		q = p->next;
		free(p);
		p = q;
	}
	free(queue);
}
int main(void){
	Queue *queue = getQueue();
	for(int i = 0; i < 10; i++){
		push(queue, i);
	}
 	printfQueue(queue);
 	printf("队列的长度：%d", getSize(queue));
 	printf("\n");
 	for(int i = 0; i < 3; i++){
 		if(pop(queue)){
 			printfQueue(queue);
 			printf("队列的长度：%d", getSize(queue));
		}
		printf("\n");
	 }
	clear(queue);
	return 0;
}
